# 핵심 로직
배열 정렬 후 아래 후위 순회만 적용해주면 끝나는 문제
```java
static void postorder(int[] arr, int left, int right) {
    if (left > right) {
        return;
    }

    int mid = (left + right) / 2; // 가운데 원소가 루트

    // 왼쪽
    postorder(arr, left, mid - 1);
    // 오른쪽
    postorder(arr, mid + 1, right);
    // 루트
    sb.append(arr[mid]).append(' ');
}
```
